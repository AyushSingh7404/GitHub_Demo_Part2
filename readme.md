# Git Branching & Merging

git init

git add .

git commit -m "Commit name"

git log --all --graph

git branch branch_name      # command to create new branch

git checkout branch_name     # command to switch to any branch

git merge branch_name_which_you_want_to_merge_with_current_branch -m "merge commit message"   # the merge is stored in a commit

git push origin branch_name_you_want_to_push

git fetch

git pull origin master
